1.查詢某個客戶編號在某一日購買的產品項目

SELECT p.prodno, p.prodname,m.orddate FROM product p, myorder m, myorderitem mi WHERE m.ordno=mi.ordno AND mi.prodno=p.prodno AND m.cusno='001' AND m.orddate='20190601'

2.查詢客戶編號001在20190601購買的總金額
SELECT SUM(mi.prodqty*mi.prodprice) as '購買金額' FROM myorderitem mi, myorder m WHERE mi.ordno=m.ordno AND m.cusno='001' AND m.orddate='20190601'

3.查詢指定年度月份與指定產品編號的金額總計
SELECT SUM(mi.prodprice*mi.prodqty) FROM myorderitem mi, myorder m
WHERE mi.ordno=m.ordno
AND substr(m.orddate,1,6)='201906'
AND mi.prodno='001'

4.查詢指定年度月份與指定產品編號產生總額並使用cuszipcode欄位來分組顯示結果
SELECT SUM(ordtotal) FROM myorder m, customer c
WHERE m.cusno=c.cusno
AND substr(m.orddate,1,6)='201906'
GROUP BY c.cuszipcode

5.指定月份欄位來分割群組結果
SELECT substr(orddate,1,6) as 月份, SUM(ordtotal) AS '總計' FROM myorder GROUP BY 月份
