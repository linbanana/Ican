1.建立資料庫
create database myorder3 character set utf8;

2-1.建立customer資料表
create table customer (
cusno char(5) not null,
cusname char(10),
cuszipcode char(3),
cusaddress varchar(50),
custel char(10),
cusfax char(10),
cuscontact char(10),
primary key (cusno));

2-2.建立employee資料表
create table employee (
empno char(5) not null,
empname char(10),
deptno char(3),
emptitle char(10),
primary key (empno));

2-3.建立department資料表
create table department (
deptno char(3) not null,
deptname char(10),
primary key (deptno));

2-4.建立product資料表
create table product (
prodno char(3) not null,
prodname char(10),
prodprice int(4),
prodamount int(5),
primary key (prodno));

2-5.建立myorder資料表
create table myorder (
ordno char(5) not null,
cusno char(5) not null,
orddate char(8),
ordtotal int(10),
salesno char(5),
assistno char(5),
primary key (ordno));

2-6.建立myorderitem資料表
create table myorderitem (
ordno char(5) not null,
serial char(5) not null,
prodno char(3),
prodqty int(6),
prodprice int(10),
primary key (ordno,serial));
